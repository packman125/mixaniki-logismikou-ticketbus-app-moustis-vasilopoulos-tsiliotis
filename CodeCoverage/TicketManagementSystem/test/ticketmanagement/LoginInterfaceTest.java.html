package ticketmanagement;

import static org.junit.Assert.*;

import java.sql.Connection;

import org.junit.Test;


public class LoginInterfaceTest {

	@Test
	public void testActionPerformed() {
		LoginInterface newLoginInterface= new LoginInterface();
    	newLoginInterface.setupFrame();
    	newLoginInterface.setUsername("pete");
    	newLoginInterface.setPassword("petelian");

		newLoginInterface.getLoginButton().doClick();
		
		assertTrue((newLoginInterface.getWrongDetails()).equals(""));
	}

	@Test
	public void testFindUserinDB() {
		LoginInterface newLoginInterface= new LoginInterface();
    	newLoginInterface.setupFrame();
    	Connection con = DatabaseInteraction.connect();
		
		User registeredUser= new User("Παναγιώτης", "Τσουτσουλιανούδης", "pete", "petelian");

		assertTrue((newLoginInterface.findUserinDB(con, "pete").getPassword()).equals(registeredUser.getPassword()));
		//close DB connection
	DatabaseInteraction.disconnect(con);
	}

}
